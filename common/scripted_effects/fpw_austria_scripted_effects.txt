AUS_increse_hungraian_attitude_effect = {
	IF = {
		limit = {
			has_idea = AUS_hungary_hostile
		}
		swap_ideas = {
			remove_idea = AUS_hungary_hostile
			add_idea = AUS_hungary_neutral
		}
	}
	ELSE_IF = {
		limit = {
			has_idea = AUS_hungary_neutral
		}
		swap_ideas = {
			remove_idea = AUS_hungary_neutral
			add_idea = AUS_hungary_friendly
		}
	}
	ELSE_IF = {
		limit = {
			has_idea = AUS_hungary_friendly
		}
		swap_ideas = {
			remove_idea = AUS_hungary_friendly
			add_idea = AUS_hungary_loyal
		}
	}
	ELSE = {
		add_political_power = 50
	}
}
AUS_decrese_hungraian_attitude_effect = {
