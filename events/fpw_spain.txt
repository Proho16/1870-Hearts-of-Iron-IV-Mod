				}
			}
			modifier = {
				add = 30

					target = SPR
					value > 30
				}
			}
			modifier = {
				factor = 0
				AND = {
					is_historical_focus_on = yes
					original_tag = PRS
				}
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = FRA
				}
			}
			SPR = {
				set_country_flag = SPR_french_king
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = ENG
				}
			}
			SPR = {
				set_country_flag = SPR_british_queen
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = ITA
				}
			}
			SPR = {
				set_country_flag = SPR_italian_king
			}
		}
		if = {
			limit = {
				ROOT = {
					original_tag = PRS
				}
			}
			PRS = {
				complete_national_focus = PRS_leopold_on_the_spanish_thrown
			}
			SPR = {
				set_country_flag = SPR_prussian_king
			}
		}
		SPR = {
			country_event = spain_fpw.3
		}
	}

	# Decline
	option = {
		name = spain_fpw.2.b
		if = {
			limit = {
				original_tag = PRS
			}
			custom_effect_tooltip = fpw_historical_option_tt
		}
		ai_chance = {
			base = 10 
			modifier = {
				factor = 5
				has_opinion = {
					target = SPR
					value < 1
				}
			}
			modifier = {
				factor = 0
				AND = {
					is_historical_focus_on = yes
					NOT = {
						original_tag = PRS
					}
				}
			}
			modifier = {
				add = 100
				AND = {
					is_historical_focus_on = yes
					original_tag = PRS
				}
			}
		}
		if = {
			limit = {
				original_tag = PRS
			}
			set_global_flag = fpw_declined_leopold
		}
		SPR = {
			country_event = spain_fpw.4
		}
	}
}
