	id = fpw_austria.5
	title = fpw_austria.5.t
	desc = fpw_austria.5.d
	picture = GFX_report_event_gathering_protest
	is_triggered_only = yes
	fire_only_once=yes
	option = {
		name = fpw_austria.5.a
		set_country_flag = hungarian_protests
		add_stability = -0.05
		add_war_support = -0.05
		hidden_effect = {
			every_owned_state = {
				limit = {
					is_core_of = HUN
					is_core_of = AUS
				}
				remove_core_of = AUS
				start_resistance = yes
				set_resistance = 60
				set_compliance = 20
			}
			country_event = {
				id = fpw_austria.6
				days = 30
			}
			remove_ideas = {
				AUS_hungary_hostile
				AUS_hungary_neutral
				AUS_hungary_friendly
				AUS_hungary_loyal
			}
			add_ideas = AUS_hungary_hostile
		}
		custom_effect_tooltip = hungarian_protests_effect
	}
}
